编程：在屏幕中间分别显示绿色、绿色红底、白色蓝底的字符串’welcome to masm!’

assume cs:code,ds:data
data segment
    db "welcome to masm!"
data ends

code segment
start:    mov ax,data
        mov ds,ax
        mov ax,0b800h
        mov es,ax

        mov bx,0
        mov si,1980
        mov cx,16
s:      mov al,ds:[bx]
        mov ah,02h
        mov es:[si],ax

        mov ah,24h
        mov es:[si+160],ax

        mov ah,71h
        mov es:[si+320],ax

        inc bx
        add si,2
        loop s

        mov ax,4c00h
        int 21h
code ends

end start
